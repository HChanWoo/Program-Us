### 푼 사람
[황찬우 풀이]
```python
def solution(command):

    point = [0,0] # 현재 위치
    dir = 0 # 방향
    move = [[0,1],[1,0],[0,-1],[-1,0]] # dir별 이동

    for ele in command :
        if ele == 'R' :
            dir+=1
        elif ele == 'L' :
            dir-=1
        elif ele == 'G' :
            point[0]+=move[dir][0]
            point[1]+=move[dir][1]
        else :
            point[0]-=move[dir][0]
            point[1]-=move[dir][1]
        if dir < 0 :
            dir=3
        elif dir > 3 :
            dir=0

    return point
```
